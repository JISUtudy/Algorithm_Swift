func solution(_ board:[[Int]], _ moves:[Int]) -> Int {
  var result = 0 
  var boards = board
  var basket = [Int]()

  for i in moves {
    for index in 0..<boards.count {
      if boards[index][i-1] != 0 {
        if basket.last == boards[index][i-1] {
          result += 2
          basket.removeLast()
        } else {
          basket.append(boards[index][i-1])
        }
        boards[index][i-1] = 0
        break
      }
    }
  }
  return result
}
