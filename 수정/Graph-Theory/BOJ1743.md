let input = readLine()!.split(separator: " ").map {Int(String($0))!}
let (n, m, k) = (input[0], input[1], input[2])

for _ in 0..<k {
    let input = readLine()!.split(separator: " ").map {Int(String($0))!}
    let (r, c) = (input[0], input[1])
    graph[r-1][c-1] = 1
}

var dx = [1, 0, -1, 0]
var dy = [0, 1, 0, -1]

func bfs(_ x: Int, _ y: Int) -> Int{
    var queue = [(x, y)]
    var cnt = 0
    while !queue.isEmpty {
        let (x, y) = queue.removeFirst()
        cnt += 1
        for i in 0..<4 {
            let nx = x + dx[i], ny = y + dy[i]
            if nx < 0 || nx >= n || ny < 0 || ny >= m {continue}
            if graph[nx][ny] == 1 {
                queue.append((nx, ny))
                graph[nx][ny] = 0
            }
        }
    }
    return cnt
}

var result = 0
for i in 0..<n {
    for j in 0..<m {
        if graph[i][j] == 1 {
            graph[i][j] = 0
            result = max(bfs(i, j), result)
        }
    }
}
print(result)
