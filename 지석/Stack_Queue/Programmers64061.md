func solution(_ board:[[Int]], _ moves:[Int]) -> Int {
    var my_list = [Int]()
    var my_board = board
    var remove_cnt: Int = 0
    
    for index in moves {
        for (i, value) in my_board.enumerated() {
            if value[index-1] != 0 {
                my_list.append(value[index-1])
                my_board[i][index-1] = 0
                break
            }
        }
        
        if my_list.count >= 2 {
            if my_list[my_list.endIndex-1] == my_list[my_list.endIndex-2] {
                my_list.removeLast()
                my_list.removeLast()
                remove_cnt += 2
            }
        }
    }
    
    return remove_cnt
}
